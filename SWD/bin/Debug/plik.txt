%wyznacza dostepne cele leczenia
moze_leczyc(bol_glowy) :- ma(bol_glowy).
moze_leczyc(bol_zeba) :- ma(bol_zeba).
moze_leczyc(goraczka) :- ma(goraczka).
moze_leczyc(bole_miesni) :- ma(bole_miesni).
moze_leczyc(biegunka) :- ma(biegunka).

%cel leczenia bedzie dostepny do wyboru na podstawie wartosci moze_leczyc()
%cel_leczenia(zmniejszenie_bolu_glowy).
%cel_leczenia(zbicie_temperatury).
cel_leczenia(zmniejszenie_bolu_zeba).
cel_leczenia(zmniejszenie_bolu_miesni).
cel_leczenia(eliminuj_biegunke).

przepisz(lek_z_paracetamolem) :-
(cel_leczenia(zmniejszenie_bolu_glowy);cel_leczenia(zmniejszenie_bolu_zeba);cel_leczenia(zbicie_temperatury)), 
not(ma(uczulenie_na_paracetamol)).

przepisz(lek_z_loperamidem) :- 
(cel_leczenia(zmniejszenie_bolu_glowy);cel_leczenia(zmniejszenie_bolu_zeba);cel_leczenia(zbicie_temperatury);cel_leczenia(zmniejszenie_bolu_miesni)), 
not(ma(uczulenie_na_loperamid)).

przepisz(pij_duzo_plynow) :- 
cel_leczenia(eliminuj_biegunke).

co_wziac(L) :- findall(X, przepisz(X), L).
%co_leczyc(L) :- findall(X, moze_leczyc(X), L).
